import os

def grant_execute_permission(filename):
    """
    Grant execute permission for the current user to the specified file.
    
    Parameters:
        filename (str): The name of the file to grant execute permission.
        
    Returns:
        bool: True if the permission was successfully granted, False otherwise.
    """
    try:
        # Get the current file permissions
        current_permissions = os.stat(filename).st_mode
        
        # Add the execute permission for the owner
        new_permissions = current_permissions | 0o100
        
        # Set the new permissions for the file
        os.chmod(filename, new_permissions)
        
        print(f"Execute permission granted for {filename} to the current user.")
        return True
    except Exception as e:
        print(f"Error granting execute permission for {filename}: {e}")
        return False

# Example Usage
filename = "your_file_to_grant_permission"
grant_execute_permission(filename)
