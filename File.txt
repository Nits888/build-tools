# Sample JSON data
response_json = {'workflows': [{'name': '{{config_name}}', 'id': 123, 'url': 'https://abc.com'},
                               {'name': 'xyz', 'id': 456, 'url': 'https://xyz.com'}]}

# Variable with the name to search for
config_name = 'abc'

# Function to replace placeholders in JSON data with values from the current scope
def replace_placeholders(json_data, current_scope):
    for key, value in json_data.items():
        if isinstance(value, str) and value.startswith("{{") and value.endswith("}}"):
            variable_name = value[2:-2]
            if variable_name in current_scope:
                json_data[key] = current_scope[variable_name]
    return json_data

# Replace placeholders in JSON data with values from the current scope
updated_json = replace_placeholders(response_json, locals())

print(updated_json)
