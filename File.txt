def fetch_git_info(repo_path):
    """Fetches pull request and release information using GitPython."""
    repo = git.Repo(repo_path)
    
    # Get the latest pull request number
    git_pull_request = repo.git.fetch("origin", "pulls")
    git_pull_request = repo.git.show_ref("--head", "--heads", "origin/pulls")
    git_pull_request = re.search(r"refs/pull/(\d+)/head", git_pull_request).group(1)
    
    # Get the latest release version
    git_release_version = repo.git.tag("--sort=-v:refname").splitlines()[0]
    
    # Get the URL of the pull request
    git_pull_request_url = f"{GIT_BASE_URL}/pull/{git_pull_request}"
    
    # Get the URL of the latest release
    git_release_url = f"{GIT_BASE_URL}/releases/tag/{git_release_version}"
    
    return git_pull_request_url, git_release_url, git_release_version
