def replace_placeholders(json_data, current_scope, env_variables):
    if isinstance(json_data, dict):
        for key, value in json_data.items():
            if isinstance(value, str):
                # Find all placeholders in the value
                placeholders = re.findall(r"\{{(.*?)\}}", value)
                for placeholder in placeholders:
                    if placeholder.strip() in current_scope:
                        # Replace the placeholder with the value from current_scope
                        value = value.replace("{{" + placeholder + "}}", str(current_scope[placeholder.strip()]))
                    elif placeholder.strip() in env_variables:
                        # Replace the placeholder with the value from env_variables
                        value = value.replace("{{" + placeholder + "}}", str(env_variables[placeholder.strip()]))
                json_data[key] = value
            elif isinstance(value, (dict, list)):
                replace_placeholders(value, current_scope, env_variables)
    elif isinstance(json_data, list):
        for i in range(len(json_data)):
            replace_placeholders(json_data[i], current_scope, env_variables)
    return json_data
